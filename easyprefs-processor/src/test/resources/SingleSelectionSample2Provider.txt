package com.nagopy.android.easyprefs.sample.prefs;

import android.app.Application;
import android.content.SharedPreferences;
import android.preference.PreferenceManager;
import java.lang.Override;
import java.lang.String;
import javax.inject.Inject;

public class GenSingleSelectionSample2 implements SingleSelectionSample2 {
  private final SharedPreferences sp;

  private final com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2.TargetEnum defValue;

  private final String key;

  @Inject
  public GenSingleSelectionSample2(Application application) {
    sp = PreferenceManager.getDefaultSharedPreferences(application);
    key = "com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2";
    defValue = com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2.TargetEnum.valueOf("TEST");
  }

  @Override
  public com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2.TargetEnum getValue() {
    String value = sp.getString(key, null);
    if (value == null || value.isEmpty()) {
      return defValue;
    }
    return com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2.TargetEnum.valueOf(value);
  }

  @Override
  public com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2.TargetEnum getDefaultValue() {
    return defValue;
  }

  @Override
  public boolean updateValue(com.nagopy.android.easyprefs.sample.prefs.SingleSelectionSample2.TargetEnum newValue) {
    return sp.edit().putString(key, newValue.name()).commit();
  }

  @Override
  public boolean clearValue() {
    return sp.edit().remove(key).commit();
  }

  @Override
  public String getKey() {
    return key;
  }
}
